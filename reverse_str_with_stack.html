step1: define an empty array and current size let data = [];
let currentSize = data.length;

step2: write push and pop funtion like that -- function push(newVal) {
data[currentSize] = newVal;
currentSize += 1
}
function pop() {
let lastRemovedItem = data[currentSize - 1];
currentSize -= 1;
data.length = currentSize;
return lastRemovedItem
}
